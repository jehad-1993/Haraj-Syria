function populateCountries() {
  var countrySelect = document.getElementById('registerCountry');
  if (!countrySelect) return;
  
  var currentCountries = countries[currentLanguage];
  countrySelect.innerHTML = '<option value="">' + translations[currentLanguage].chooseCountry + '</option>';
  
  Object.keys(currentCountries).forEach(function(code) {
    var option = document.createElement('option');
    option.value = code;
    option.textContent = currentCountries[code].name;
    countrySelect.appendChild(option);
  });
  
  countrySelect.value = 'SY';
  updateCitiesAndPhoneCode();
}

function updateCitiesAndPhoneCode() {
  var countrySelect = document.getElementById('registerCountry');
  var citySelect = document.getElementById('registerCity');
  var phoneCodeInput = document.getElementById('phoneCode');
  
  if (!countrySelect || !citySelect || !phoneCodeInput) return;
  
  var selectedCode = countrySelect.value;
  
  if (selectedCode && countries[currentLanguage][selectedCode]) {
    phoneCodeInput.value = countries[currentLanguage][selectedCode].code;
    citySelect.disabled = false;
    
    citySelect.innerHTML = '<option value="">' + translations[currentLanguage].chooseCity + '</option>';
    var cities = countries[currentLanguage][selectedCode].cities;
    cities.forEach(function(city) {
      var option = document.createElement('option');
      option.value = city;
      option.textContent = city;
      citySelect.appendChild(option);
    });
  } else {
    phoneCodeInput.value = '+963';
    citySelect.disabled = true;
    citySelect.innerHTML = '<option value="">' + translations[currentLanguage].chooseCity + '</option>';
  }
}

function checkPasswordRequirements() {
  var passwordField = document.getElementById('registerPassword');
  if (!passwordField) return;
  
  var password = passwordField.value;
  var requirements = {
    length: password.length >= 8,
    uppercase: /[A-Z]/.test(password),
    lowercase: /[a-z]/.test(password),
    number: /[0-9]/.test(password),
    special: /[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]/.test(password)
  };
  
  Object.keys(requirements).forEach(function(req) {
    var element = document.getElementById('req-' + req);
    var icon = element.querySelector('.requirement-icon');
    if (requirements[req]) {
      element.classList.add('valid');
      element.classList.remove('invalid');
      icon.textContent = '✓';
    } else {
      element.classList.add('invalid');
      element.classList.remove('valid');
      icon.textContent = '○';
    }
  });
}

function toggleLanguage() {
  currentLanguage = currentLanguage === 'ar' ? 'en' : 'ar';
  updateLanguage();
}

function updateLanguage() {
  var isArabic = currentLanguage === 'ar';
  
  document.documentElement.setAttribute('lang', currentLanguage);
  document.documentElement.setAttribute('dir', isArabic ? 'rtl' : 'ltr');
  document.body.className = isArabic ? 'rtl bg-gray-50' : 'ltr bg-gray-50';
  
  Object.keys(translations[currentLanguage]).forEach(function(key) {
    var element = document.getElementById(key);
    if (element) {
      element.textContent = translations[currentLanguage][key];
    }
  });
  
  var searchInput = document.getElementById('searchInput');
  if (searchInput) {
    searchInput.placeholder = translations[currentLanguage].searchPlaceholder;
  }
  
  populateCountries();
  
  var passwordField = document.getElementById('registerPassword');
  if (passwordField && passwordField.value) {
    checkPasswordRequirements();
  }
}

function showPage(pageId) {
  currentPage = pageId;
  
  var pages = document.querySelectorAll('.page');
  pages.forEach(function(page) {
    page.classList.remove('active');
  });
  
  var targetPage = document.getElementById(pageId);
  if (targetPage) {
    targetPage.classList.add('active');
  }
  
  var navLinks = document.querySelectorAll('.nav-link');
  navLinks.forEach(function(link) {
    link.classList.remove('active');
  });
  
  var navMapping = {
    'home': 'navHome',
    'categories': 'navCategories',
    'register': 'navRegister', 
    'login': 'navLogin',
    'post-ad': 'navPostAd'
  };
  
  var activeNavId = navMapping[pageId];
  if (activeNavId) {
    var activeNavLink = document.getElementById(activeNavId);
    if (activeNavLink) {
      activeNavLink.classList.add('active');
    }
  }
}

function handleRegister(event) {
  event.preventDefault();
  var name = document.getElementById('registerName').value;
  var email = document.getElementById('registerEmail').value;
  var country = document.getElementById('registerCountry').value;
  var city = document.getElementById('registerCity').value;
  var password = document.getElementById('registerPassword').value;
  var confirmPassword = document.getElementById('confirmPassword').value;
  
  if (password !== confirmPassword) {
    alert(currentLanguage === 'ar' ? 'كلمات المرور غير متطابقة!' : 'Passwords do not match!');
    return;
  }
  
  var requirements = {
    length: password.length >= 8,
    uppercase: /[A-Z]/.test(password),
    lowercase: /[a-z]/.test(password),
    number: /[0-9]/.test(password),
    special: /[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]/.test(password)
  };
  
  var allRequirementsMet = Object.values(requirements).every(function(req) {
    return req;
  });
  
  if (!allRequirementsMet) {
    alert(currentLanguage === 'ar' ? 'كلمة المرور لا تستوفي جميع المتطلبات!' : 'Password does not meet all requirements!');
    return;
  }
  
  alert(currentLanguage === 'ar' ?
    'تم إنشاء الحساب بنجاح!\nالاسم: ' + name + '\nالمدينة: ' + city :
    'Account created successfully!\nName: ' + name + '\nCity: ' + city
  );
}
